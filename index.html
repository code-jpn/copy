const params = new URLSearchParams(window.location.search); const text = params.get('s'); const anchor = params.get('a'); if (text) { navigator.clipboard.writeText(text).then(() => { console.log('コピー成功:', text); }).catch(err => { console.error('コピー失敗:', err); }); } else { console.warn('コピーするテキストがを指定してください'); } if (document.referrer) { window.location.href = document.referrer + (anchor ? `#${anchor}` : ''); } else { console.log('トップページ(仮)に戻ります。'); // window.location.href = '/'; // 必要に応じて有効化 }
